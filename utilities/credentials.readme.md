# Sample Credentials
Create a credentials.ts file
Both the credentials.ts and credentials.js will be in your .gitignore and should not be committed to github.

```json
export const Credentials = {
  userB: {
    loginName: 'UserB_Login',
    password: 'UserB_Password',
    baseUrl: 'https://yoursite.com'
  },
  michael: {
    loginName: 'Michael_Login',
    password: 'Michael_Password',
     baseUrl: 'https://yoursite.com'
  },
  superUser: {
    loginName: 'SU_Login',
    password: 'SU_Password',
     baseUrl: 'https://yoursite.com'

  }
};
```
